
### `flags_ex1.asm` – תרגול 1: הבנת דגלים

```asm
IDEAL
MODEL small
STACK 100h

DATASEG
; no variables needed

CODESEG
start:
	mov ax, @data
	mov ds, ax

	; example 1 - ZF
	mov ax, 5
	sub ax, 5       ; AX = 0 → ZF = 1

	; example 2 - SF
	mov ax, 10
	sub ax, 20      ; AX = -10 → SF = 1

	; example 3 - CF
	mov ax, 0FFFFh
	add ax, 1       ; AX = 0 → CF = 1, ZF = 1

exit:
	mov ax, 4c00h
	int 21h
END start
```

### `flags_ex2.asm` – תרגול 2: תנאים ו־CMP

```asm
IDEAL
MODEL small
STACK 100h

DATASEG
equal   db ?
greater db ?
less    db ?
zero_flagged db ?
sign_status  db ?

CODESEG
start:
	mov ax, @data
	mov ds, ax

	; ===== comparing two numbers =====
	mov ax, 10
	mov bx, 20
	cmp ax, bx
	je eq_case
	ja gt_case
	jb lt_case

eq_case:
	mov equal, 1
	jmp done_cmp
gt_case:
	mov greater, 2
	jmp done_cmp
lt_case:
	mov less, 3
done_cmp:

	; ===== comparison against zero =====
	mov al, 0
	cmp al, 0
	jz is_zero
	mov zero_flagged, 0
	jmp after_zero
is_zero:
	mov zero_flagged, 1
after_zero:

	; ===== sign check =====
	mov ax, -1
	cmp ax, 0
	js negative
	mov sign_status, 0
	jmp done_sign
negative:
	mov sign_status, -1
done_sign:

exit:
	mov ax, 4c00h
	int 21h
END start
```

### `flags_ex3.asm` – תרגול 3: לולאות

```asm
IDEAL
MODEL small
STACK 100h

DATASEG
counter    db 0
done_value db ?

CODESEG
start:
	mov ax, @data
	mov ds, ax

	; ===== loop 10 times =====
	mov cx, 10
loop1:
	mov al, counter
	inc al
	mov counter, al
	loop loop1

	; ===== loop from 20 to 0 =====
	mov cx, 21
	mov al, 20
loop2:
	dec al
	loop loop2
	mov done_value, al

	; ===== loop while AL is odd =====
	mov al, 7
odd_loop:
	test al, 1
	jz end_odd
	dec al
	jmp odd_loop
end_odd:

exit:
	mov ax, 4c00h
	int 21h
END start
```

### `flags_ex4.asm` – תרגול 4: בדיקות לפי דגלים

```asm
IDEAL
MODEL small
STACK 100h

DATASEG
overflow_happened db ?
is_negative       db ?
not_equal_flag    db ?

CODESEG
start:
	mov ax, @data
	mov ds, ax

	; ===== overflow check =====
	mov al, 100
	add al, 200
	jo ovf
	jmp continue_ovf
ovf:
	mov overflow_happened, 1
continue_ovf:

	; ===== negative result =====
	mov ax, -10
	cmp ax, 0
	js neg_case
	jmp continue_sign
neg_case:
	mov is_negative, 1
continue_sign:

	; ===== inequality =====
	mov ax, 5
	mov bx, 6
	cmp ax, bx
	jne noteq
	jmp done_noteq
noteq:
	mov not_equal_flag, 99
done_noteq:

exit:
	mov ax, 4c00h
	int 21h
END start
```

### `flags_ex5.asm` – תרגול 5: תנאים מתקדמים, TEST וקפיצות

```asm
IDEAL
MODEL small
STACK 100h

DATASEG
grade         db 87
result_code   db ?
carry_status  db ?
test_result   db ?

CODESEG
start:
	mov ax, @data
	mov ds, ax

	; ===== grade simulation =====
	mov al, grade
	cmp al, 60
	jb failed
	cmp al, 85
	ja excellent
	jmp passed

failed:
	mov result_code, 0  ; failed
	jmp after_grade
passed:
	mov result_code, 1  ; passed
	jmp after_grade
excellent:
	mov result_code, 2  ; excellent
after_grade:

	; ===== check if there's no carry =====
	mov al, 0FFh
	add al, 1
	jnc no_carry
	mov carry_status, 1
	jmp done_carry
no_carry:
	mov carry_status, 0
done_carry:

	; ===== check with TEST =====
	mov al, 5
	test al, al
	jnz non_zero
	mov test_result, 0
	jmp end_test
non_zero:
	mov test_result, 1
end_test:

exit:
	mov ax, 4c00h
	int 21h
END start
```

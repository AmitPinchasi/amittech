### פתרון לתרגיל 1

```c
#include <stdio.h>

int main() {
    char name[100];
    int age;
    float height;

    printf("הכנס שם, גיל וגובה:\n");
    scanf("%s %d %f", name, &age, &height);

    printf("שלום %s! אתה בן %d וגובהך %.2f מטרים\n", name, age, height);
    return 0;
}
```

### פתרון לתרגיל 2

```c
#include <stdio.h>
#include <string.h>

int main() {
    char input[100];
    printf("הכנס שם משתמש:\n");
    scanf("%s", input);

    if (strncmp(input, "admin", 100) == 0) {
        puts("ברוך הבא מנהל");
    } else {
        puts("גישה נדחתה");
    }

    return 0;
}
```

    1.3 - ОљОќ ОъОћ ОќОћ ОЉОљОЕ - ОцОфОеОЋОЪ

    ОфОеОњОЎОюОЎОЮ¶

    ОљОќ ОъОћ ОќОћ ОЉОљОЕ?¶

    ОцОфОеОЋОЪ:¶

    bash ОћОЎОљ ОЕОцОф ОАОДОеОЎОцОўОЎОЮ ОЋ-shell ОЕОъОљОцОЕОеОф ОюОаОЋ ОюОфОДОЕОе ОбОЮ ОъОбОеОЏОф ОћОћОцОбОюОћ. Оюbash ОЎОЕ ОцОДОЋОЊОЋОф ОъОЋОЉОаОЋОф (built-in) ОЕОћОЪ ОЌОюОД Оъbash ОбОдОъОћ, ОЋОЎОЕ ОцОДОЋОЊОЋОф ОЌОЎОдОЋОаОЎОЋОф ОЕОћОЪ ОфОЋОЏОаОЎОЋОф ОбОдОъОљОЎОЋОф ОЕОъОЋОфОДОаОЋОф ОЉОъОбОеОЏОф.

    ОцОДОЋОЊОЋОф ОъОЋОЉОаОЋОф ОЕОю bash - ОюОЊОЋОњОъОћ:
    - cd - ОЕОЎОаОЋОЎ ОфОЎОДОЎОЎОћ
    - echo - ОћОЊОцОАОћ ОюОъОАОџ
    - export - ОћОњОЊОеОф ОъОЕОфОаОЎ ОАОЉОЎОЉОћ
    - alias - ОЎОдОЎОеОф ОДОЎОдОЋОеОЎ ОцОДОЋОЊОЋОф
    - history - ОћОдОњОф ОћОЎОАОўОЋОеОЎОЎОф ОцОДОЋОЊОЋОф

    ОЏОЊОЎ ОюОЉОЊОЋОД ОљОЮ ОцОДОЋОЊОћ ОћОЎОљ ОъОЋОЉОаОЎОф ОЕОю bash ОљОЋ ОЌОЎОдОЋОаОЎОф:

    type ls
# ls is /usr/bin/ls  -- external command

type cd
# cd is a shell builtin  -- builtin command

type echo
# echo is a shell builtin  -- builtin command

    ОЏОЊОЎ ОюОеОљОЋОф ОљОф ОћОаОфОЎОЉ ОћОъОюОљ ОЕОю ОцОДОЋОЊОћ ОЌОЎОдОЋОаОЎОф:

    which ls
# /usr/bin/ls

which cat
# /usr/bin/cat

which python3
# /usr/bin/python3

    ОцОДОЋОЊОЋОф ОЌОЎОдОЋОаОЎОЋОф ОЕОљОЎОаОЪ ОДОЕОЋОеОЋОф Ою-bash:

    which curl
# /usr/bin/curl

man curl
# shows that curl is a program for downloading files from the internet

which git
# /usr/bin/git

man git
# shows that git is a version control system

    ОћОАОЉОЎОЉОћ ОЕОюОЎ¶

    ОцОфОеОЋОЪ:¶

    ОћОдОњОф ОЏОю ОъОЕОфОаОЎ ОћОАОЉОЎОЉОћ ОбОЮ env:

    env

    ОћОдОњОф ОъОЕОфОаОЎ ОАОЉОЎОЉОћ ОЋОцОЋОаОДОдОЎОЋОф bash ОбОЮ set:

    set

    ОЏОъОћ ОъОЕОфОаОЎ ОАОЉОЎОЉОћ ОЌОЕОЋОЉОЎОЮ ОЋОюОъОћ ОћОЮ ОЕОЎОъОЋОЕОЎОЎОЮ:

    • PATH - ОеОЕОЎОъОф ОћОфОЎОДОЎОЋОф ОЕОЉОћОЪ bash ОъОЌОцОЕ ОцОДОЋОЊОЋОф. ОЏОЕОЏОЋОфОЉОЎОЮ ls bash ОбОЋОЉОе ОбОю ОЏОю ОфОЎОДОЎОћ ОЉ-PATH ОбОЊ ОЕОъОЋОдОљ ОљОф ОћОДОЋОЉОЦ ls.
    echo $PATH
# /usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin
    • HOME - ОфОЎОДОЎОЎОф ОћОЉОЎОф ОЕОю ОћОъОЕОфОъОЕ ОћОаОЋОЏОЌОЎ. ОЏОЕОЏОЋОфОЉОЎОЮ cd ~ bash ОЎОЋОЊОб ОюОљОЪ ОюОюОЏОф ОЉОќОЏОЋОф ОћОъОЕОфОаОћ ОћОќОћ.
    echo $HOME
# /home/ubuntu
    • USER - ОЕОЮ ОћОъОЕОфОъОЕ ОћОаОЋОЏОЌОЎ. ОЕОЎОъОЋОЕОЎ ОЉОАОДОеОЎОцОўОЎОЮ ОЕОдОеОЎОЏОЎОЮ ОюОЊОбОф ОъОЎ ОъОеОЎОЦ ОљОЋОфОЮ.
    echo $USER
# ubuntu
    • SHELL - ОћОаОфОЎОЉ Ою-shell ОЕОћОъОЕОфОъОЕ ОъОЕОфОъОЕ ОЉОЋ.
    echo $SHELL
# /bin/bash
    • LANG - ОћОњОЊОеОф ОћОЕОцОћ ОЋОћОДОЎОЊОЋОЊ ОЕОю ОћОъОбОеОЏОф. ОъОЕОцОЎОб ОбОю ОљОЎОџ ОфОЋОЏОаОЎОЋОф ОъОдОЎОњОЋОф ОўОДОАОў.
    echo $LANG
# en_US.UTF-8
    • PWD - ОфОЎОДОЎОЎОф ОћОбОЉОЋОЊОћ ОћОаОЋОЏОЌОЎОф (Present Working Directory). ОъОфОбОЊОЏОЪ ОљОЋОўОЋОъОўОЎОф ОЉОЏОю cd.
    echo $PWD
# /home/ubuntu

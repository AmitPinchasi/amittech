    תרגיל 1 - DRY¶

    def generate_task_report_for_manager(tasks: list) -> str:
    report = "Task report\n"
    report += "=" * 30 + "\n"
    overdue = [t for t in tasks if t["status"] != "done" and t["days_left"] < 0]
    report += f"Overdue tasks: {len(overdue)}\n"
    for task in overdue:
        report += f"  - {task['title']}: {abs(task['days_left'])} days overdue\n"
    return report


def generate_task_report_for_developer(tasks: list) -> str:
    report = "Task report\n"
    report += "=" * 30 + "\n"
    overdue = [t for t in tasks if t["status"] != "done" and t["days_left"] < 0]
    report += f"Overdue tasks: {len(overdue)}\n"
    for task in overdue:
        report += f"  - {task['title']}: {abs(task['days_left'])} days overdue\n"
    in_progress = [t for t in tasks if t["status"] == "in_progress"]
    report += f"\nIn-progress tasks: {len(in_progress)}\n"
    return report


def generate_task_report_for_client(tasks: list) -> str:
    report = "Task report\n"
    report += "=" * 30 + "\n"
    done = [t for t in tasks if t["status"] == "done"]
    report += f"Completed tasks: {len(done)}\n"
    overdue = [t for t in tasks if t["status"] != "done" and t["days_left"] < 0]
    report += f"Overdue tasks: {len(overdue)}\n"
    return report

    תרגיל 2 - KISS¶

    הקוד הבא מבצע חישוב פשוט אבל בצורה מסובכת מדי. פשטו אותו.

    from functools import reduce
from typing import Callable

def calculate_task_completion_rate(tasks: list) -> float:
    pipeline: list[Callable] = [
        lambda t: filter(lambda x: x.get("status") is not None, t),
        lambda t: list(t),
        lambda t: (
            reduce(lambda acc, x: acc + (1 if x["status"] == "done" else 0), t, 0),
            len(t)
        )
    ]
    done_count, total = reduce(lambda data, fn: fn(data), pipeline, tasks)
    return (done_count / total * 100) if total > 0 else 0.0

    class CommentSystem:
    def __init__(self):
        self.comments = []
        self.deleted_comments = []  # for a future recycle bin
        self.comment_versions = {}  # for historical edits

    def add_comment(self, task_id: int, user_id: int, content: str):
        comment = {
            "id": len(self.comments) + 1,
            "task_id": task_id,
            "user_id": user_id,
            "content": content,
            "version": 1,
            "translations": {},       # for future translation
            "reactions": {},          # for future like/dislike
            "thread_parent_id": None  # for future threaded comments
        }
        self.comments.append(comment)
        self.comment_versions[comment["id"]] = [comment.copy()]
        return comment

    def get_comments(self, task_id: int):
